配置 persistence.xml
```

...
```

```
@Singleton
public class UserEaoImpl implements UserEao {
	@PersistenceUnit(name = "primary")
	EntityManagerFactory emf;

	@Override
	public User find(Integer id) {
		EntityManager em = emf.createEntityManager();
		User user = em.find(User.class, id);
		em.close();
		return user;
	}
}
```
